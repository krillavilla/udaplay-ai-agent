10   ### Setup 
11   #%% 
12   # Only needed for Udacity workspace
13   
14   import importlib.util
15   import sys
16   
17   # Check if 'pysqlite3' is available before importing
18   if importlib.util.find_spec("pysqlite3") is not None:
19       import pysqlite3
20       sys.modules['sqlite3'] = sys.modules.pop('pysqlite3')
21   #%% 
22   import os
23   import json
24   import chromadb
25   from chromadb.utils import embedding_functions
26   from dotenv import load_dotenv
27   
28   # Load environment variables
29   load_dotenv()
30   
31   print("Libraries imported successfully!")
32   #%% 
33   # Initialize ChromaDB client with persistent storage
34   chroma_client = chromadb.PersistentClient(path="chromadb")
35   
36   # Set up OpenAI embedding function
37   embedding_fn = embedding_functions.OpenAIEmbeddingFunction(
38       api_key=os.getenv("OPENAI_API_KEY"),
39       api_base="https://openai.vocareum.com/v1",
40       model_name="text-embedding-ada-002"
41   )
42   
43   print("ChromaDB client and embedding function initialized!")
44   #%% 
45   # Create or get collection
46   collection_name = "udaplay"
47   
48   try:
49       # Try to get existing collection first
50       collection = chroma_client.get_collection(
51           name=collection_name,
52           embedding_function=embedding_fn
53       )
54       print(f"Retrieved existing collection: {collection_name}")
55       print(f"Current document count: {collection.count()}")
56   except:
57       # Create new collection if it doesn't exist
58       collection = chroma_client.create_collection(
59           name=collection_name,
60           embedding_function=embedding_fn
61       )
62       print(f"Created new collection: {collection_name}")
63   #%% md 
64   ### Data Loading 
65   #%% 
66   # Load game data from JSON files
67   data_dir = "games"
68   
69   # Check if directory exists
70   if not os.path.exists(data_dir):
71       print(f"Directory '{data_dir}' not found!")
72       print("Please ensure the games directory exists with JSON files.")
73   else:
74       json_files = [f for f in os.listdir(data_dir) if f.endswith('.json')]
75       print(f"Found {len(json_files)} JSON files in '{data_dir}' directory")
76       
77       # Show first few files
78       for i, filename in enumerate(sorted(json_files)[:5]):
79           print(f"   {i+1}. {filename}")
80       
81       if len(json_files) > 5:
82           print(f"   ... and {len(json_files) - 5} more files")
83   #%% 
84   # Add documents to ChromaDB
85   if os.path.exists(data_dir):
86       documents_added = 0
87       
88       for file_name in sorted(os.listdir(data_dir)):
89           if not file_name.endswith(".json"):
90               continue
91               
92           file_path = os.path.join(data_dir, file_name)
93           
94           try:
95               with open(file_path, "r", encoding="utf-8") as f:
96                   game = json.load(f)
97               
98               # Create content string for embedding
99               content = f"[{game['Platform']}] {game['Name']} ({game['YearOfRelease']}) - {game['Description']}"
100              
101              # Use filename (without extension) as document ID
102              doc_id = os.path.splitext(file_name)[0]
103              
104              # Add to collection
105              collection.add(
106                  ids=[doc_id],
107                  documents=[content],
108                  metadatas=[game]
109              )
110              
111              documents_added += 1
112              
113          except Exception as e:
114              print(f"Error processing {file_name}: {e}")
115      
116      print(f"Successfully added {documents_added} documents to ChromaDB")
117      print(f"Total documents in collection: {collection.count()}")
118  else:
119      print("Games directory not found. Cannot load data.")

121  ### Semantic Search Demonstration 
122   
123  Now that we have loaded all the game data into our vector database, let's demonstrate that semantic search works properly. 
124  #%% 
125  # Test semantic search functionality
126  def test_semantic_search(collection, query, n_results=3):
127      """Test semantic search and display results"""
128      print(f"\n=== Searching for: '{query}' ===")
129      
130      results = collection.query(
131          query_texts=[query],
132          n_results=n_results,
133          include=['documents', 'metadatas', 'distances']
134      )
135      
136      if results['documents'][0]:
137          print(f"Found {len(results['documents'][0])} results:")
138          
139          for i, (doc, metadata, distance) in enumerate(zip(
140              results['documents'][0], 
141              results['metadatas'][0], 
142              results['distances'][0]
143          )):
144              similarity = 1 - distance  # Convert distance to similarity
145              print(f"\n{i+1}. {metadata['Name']} ({metadata['YearOfRelease']})")
146              print(f"   Platform: {metadata['Platform']}")
147              print(f"   Genre: {metadata.get('Genre', 'N/A')}")
148              print(f"   Publisher: {metadata.get('Publisher', 'N/A')}")
149              print(f"   Similarity: {similarity:.3f}")
150              print(f"   Description: {metadata['Description'][:100]}...")
151      else:
152          print("No results found.")
153      
154      return results
155  
156  # Test with different types of queries
157  test_queries = [
158      "Nintendo racing games",
159      "RPG games with fantasy themes",
160      "PlayStation games from the 1990s",
161      "Action games with shooting mechanics",
162      "Games suitable for families"
163  ]
164  
165  for query in test_queries:
166      test_semantic_search(collection, query, n_results=2)
