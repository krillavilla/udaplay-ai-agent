87   #### Retrieve Game Tool 
88   #%% 
89   @tool
90   def retrieve_game_info(query: str, n_results: int = 3) -> list[str]:
91       """ 
92       Semantic search: Finds most relevant games in the vector database. 
93        
94       Args: 
95           query: A question about the game industry 
96           n_results: Number of results to return (default: 3) 
97        
98       Returns: 
99           List of JSON strings containing game metadata including: 
100          - Platform: Game Boy, PlayStation 5, Xbox 360, etc. 
101          - Name: Name of the game 
102          - YearOfRelease: Year when the game was released 
103          - Description: Additional details about the game 
104          - Genre: Game genre 
105          - Publisher: Game publisher 
106      """
107      print(f"🔍 Tool: Retrieving game info for query: '{query}'")
108      chroma_client = chromadb.PersistentClient(path="chromadb")
109  
110      embedding_fn = embedding_functions.OpenAIEmbeddingFunction(
111          api_key=os.getenv("OPENAI_API_KEY"),
112          api_base="https://openai.vocareum.com/v1",  # For Vocareum
113          model_name="text-embedding-ada-002"
114      )
115  
116      collection = chroma_client.get_collection(
117          name="udaplay",
118          embedding_function=embedding_fn
119      )
120  
121      # Query the collection
122      results = collection.query(query_texts=[query], n_results=n_results)
123  
124      # Format the retrieved documents for the next step
125      retrieved_docs = []
126      if results and results['metadatas'][0]:
127          for meta in results['metadatas'][0]:
128              # Pass the full JSON metadata as a string
129              retrieved_docs.append(json.dumps(meta))
130      
131      print(f"Retrieved {len(retrieved_docs)} games from vector database")
132      return retrieved_docs

134  #### Evaluate Retrieval Tool 
135  #%% 
136  @tool
137  def evaluate_retrieval(query: str, context: list[str]) -> bool:
138      """ 
139      Based on the user's question and retrieved documents, analyzes if the documents  
140      are sufficient to answer the question. 
141       
142      Args: 
143          query: Original question from user 
144          context: Retrieved documents most similar to the user query in the Vector Database 
145       
146      Returns: 
147          bool: True if documents are sufficient, False if web search is needed 
148      """
149      print(f"Evaluating if retrieved context is sufficient for: '{query}'")
150      
151      # A simple but effective prompt for the LLM
152      prompt = f"""
153      Based *only* on the provided context below, can you confidently answer the following user question?
154      Respond with only "yes" or "no".
155  
156      User Question: "{query}"
157  
158      Context:
159      ---
160      {context}
161      ---
162      """
163  
164      # Use the configured client object here
165      response = client.chat.completions.create(
166          model="gpt-3.5-turbo",
167          messages=[{"role": "user", "content": prompt}],
168          max_tokens=5,
169          temperature=0.0
170      )
171  
172      decision = response.choices[0].message.content.strip().lower()
173      print(f"Evaluation: LLM decision is '{decision}'")
174      return "yes" in decision

177  #%% 
178  @tool
179  def game_web_search(query: str) -> str:
180      """ 
181      Performs a web search using Tavily API when local database is insufficient. 
182       
183      Args: 
184          query: A question about the game industry 
185       
186      Returns: 
187          str: Content from the most relevant web search result 
188      """
189      print(f"🌐 Tool: Performing web search for query: '{query}'")
190      try:
191          response = tavily_client.search(query=query, search_depth="basic")
192          # We'll return the most relevant search result content
193          content = response['results'][0]['content']
194          print(f"Web search completed successfully")
195          return content
196      except Exception as e:
197          print(f"Error during Tavily search: {e}")
198          return "Web search failed."
